# CHROME i18n + Locale Plumbing — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Localize Inceptio's client UI **chrome** into de / fr / es-419 / pt-BR, and plumb a client→Worker `X-Locale` seam (accept + validate + ignore) so the later VOICE phase is a server-only change — while extracting (but **not** translating) all ruling-dependent astrology "voice" strings to `voice.*` keys.

**Architecture:** `react-i18next` + `expo-localization`, bundled JSON resources, namespace-per-file. A Foundation task installs deps, loads the FormatJS `Intl.PluralRules` polyfill before `i18n.init()`, and establishes two distinct resolutions — the **i18next bundle key** (`es-419`) and the **Intl-locale arg** (`es-419→es`). Existing `t()`-shaped seams (`card-strings`, `error-messages`) are re-pointed to i18next; remaining chrome is extracted per screen into disjoint namespace files for parallel-safe work. The Worker gains a minimal `X-Locale` shape-validator that does not affect responses or cache keys this phase.

**Tech Stack:** Expo SDK 55, RN 0.83 (New Arch, Hermes), TypeScript strict, `i18next 26.3.1`, `react-i18next 17.0.8`, `expo-localization 55.0.13`, `@formatjs/intl-pluralrules`, `@formatjs/intl-locale`, Cloudflare Worker (Vitest), Jest (mobile).

**Spec:** `docs/superpowers/specs/2026-06-08-i18n-chrome-design.md`
**Pre-flight audits:** `docs/superpowers/expert/2026-06-08-i18n-chrome-domain.md`, `docs/superpowers/library-audit/2026-06-08-i18n-chrome-design.md`

`apps/mobile/src/i18n/locale.ts`:
```ts
import { getLocales } from 'expo-localization';

export const SUPPORTED = ['en', 'de', 'fr', 'es-419', 'pt-BR'] as const;
export type Bundle = (typeof SUPPORTED)[number];

type DeviceLocale = { languageTag: string; languageCode: string | null };

/** Device preferences -> our bundle key. Explicit map; do NOT rely on i18next implicit matching. */
export function resolveLocale(locales: DeviceLocale[] = getLocales()): Bundle {
  for (const { languageTag, languageCode } of locales) {
    const primary = (languageCode ?? languageTag.split('-')[0] ?? '').toLowerCase();
    if (primary === 'es') return 'es-419'; // any Spanish -> Latam
    if (primary === 'pt') return 'pt-BR';  // any Portuguese -> Brazil
    if (primary === 'de') return 'de';
    if (primary === 'fr') return 'fr';
    if (primary === 'en') return 'en';
  }
  return 'en';
}

/** Bundle key -> Intl-valid locale arg. Hermes/ICU has no M49 `419` data. */
export function toIntlLocale(bundle: string): string {
  if (bundle === 'es-419') return 'es';
  if (bundle === 'pt-BR') return 'pt';
  return bundle;
}

let override: Bundle | null = null;
/** __DEV__ only: force a locale for the local-verification loop. */
export function __setLocaleOverride(b: Bundle | null) { override = b; }
export function activeBundle(): Bundle { return override ?? resolveLocale(); }
```

// ...one import per NS per locale for all 5 locales (en, de, fr, es-419, pt-BR).
// Full NS list (CHROME): common, nav, onboarding, activity, daterange, location,
//   loading, calendar, moment, noviable, moments, settings, paywall, today, card, errors.
// VOICE sub-files (en only): voice/card, voice/reason, voice/calendar, voice/moment, voice/moments.

const bundle = (c, nav, onboarding /* …all NS in fixed order */) => ({
  common: c, nav, onboarding, /* … */
});
const resources = {
  en: bundle(en_common, en_nav, en_onboarding /* … */),
  de: bundle(de_common, de_nav, de_onboarding /* … */),
  fr: bundle(fr_common, fr_nav, fr_onboarding /* … */),
  'es-419': bundle(es419_common, es419_nav, es419_onboarding /* … */),
  'pt-BR': bundle(ptBR_common, ptBR_nav, ptBR_onboarding /* … */),

resources.en.voice = { card: en_voice_card, reason: en_voice_reason, calendar: en_voice_calendar, moment: en_voice_moment, moments: en_voice_moments };

export function initI18n() {
  if (i18n.isInitialized) return i18n;
  i18n.use(initReactI18next).init({
    resources,
    lng: activeBundle(),
    supportedLngs: SUPPORTED,
    fallbackLng: { 'es-419': ['en'], 'pt-BR': ['en'], de: ['en'], fr: ['en'], default: ['en'] },
    load: 'currentOnly',
    keySeparator: false,
    nsSeparator: ':',
    defaultNS: 'common',
    interpolation: { escapeValue: false },
    react: { useSuspense: false },
    returnNull: false,
  });
  return i18n;
}

const LOCALE_RE = /^[A-Za-z]{2,3}(-[A-Za-z0-9]{2,8})*$/;

export function isValidLocale(raw: string | null | undefined): boolean {
